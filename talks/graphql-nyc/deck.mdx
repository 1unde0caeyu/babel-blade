import {
  // LiveProvider,
  LiveEditor,
  LiveError,
  LivePreview
} from 'react-live'
import {Header, Wrapper, Photo, Search, DogOfTheDay, ApolloProvider, ApolloClient} from './components/Demo'
import {Autocomplete, Input, Downshift} from './components/Autocomplete'

<SideBySideProvider 
  scope={{
    Header, Wrapper, Photo, Search, ApolloProvider, ApolloClient, 
    Autocomplete, Input, Downshift
    }}
  noInline={true}
  code={`
  class DogOfTheDay extends React.Component {
    constructor() {
      this.state = { selectedDog: null }
      this.onDogSelected = this.onDogSelected.bind(this)
    }
    onDogSelected(selectedDog) {
      this.setState({ selectedDog });
    };
    render() {
      return (
        <Wrapper>
          <Header>Dog of the Day! üê∂</Header>
          <Photo breed={this.state.selectedDog} />
          <Search onDogSelected={this.onDogSelected} />
        </Wrapper>
      );
    };
  }
  const client = new ApolloClient({
    uri: "https://dog-suspense-demo.glitch.me/graphql"
  })
  render(<ApolloProvider client={client}>
    <DogOfTheDay />
  </ApolloProvider>)
  `}>
    <StyledEditor />
    <LiveError />
    <LivePreview />
</SideBySideProvider>
